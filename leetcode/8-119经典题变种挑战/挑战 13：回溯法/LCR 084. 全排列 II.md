- 空间复杂度: $O()$

```python
class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        ans, n = [], len(nums)
        # vis[i] 表示 nums[i] 是否已经被选择
        vis = {i: False for i in range(n)}
        record = {}

        def dfs(i, path):
            if i == n:
                if "".join(map(str, path)) not in record:
                    ans.append(path)
                    record["".join(map(str, path))] = True
                return

            for j in range(n):
                if vis[j]:
                    continue
                vis[j] = True
                dfs(i + 1, path + [nums[j]])
                vis[j] = False
        dfs(0, [])
        return ans
```

## 2. 使用位运算


**Python3**

```python
class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        ans, n = [], len(nums)
        # vis 的第 i 位表示 nums[i] 是否被选择
        vis = 0

        nums.sort()
        def dfs(i, path):
            if i == n:
                ans.append(path)
                return

            nonlocal vis
            for j in range(n):
                if ((vis >> j) & 1) or j and nums[j] == nums[j - 1] and ((vis >> (j - 1)) & 1) == 0:
                    continue
                vis |= 1 << j
                dfs(i + 1, path + [nums[j]])
                vis &= ~(1 << j)
        dfs(0, [])
        return ans
```
